fator escala
a <- range(dados2$`Salário Mínimo`)
b <- range(dados2$Participação)

fator.escala <- diff(a)/diff(b)

dados2$y2 <- ((dados2$Participação-b[1])*fator.escala)+a[1]
transf <- ~((.-a[1])/fator.escala)+b[1]

# Criar o gráfico de linha
ggplot(dados2, aes(Data,`Salário Mínimo`))+
  geom_line(aes(col="dolar"))+
  geom_line(aes(y=y2, col="Bovespa"))
  scale_y_continuous(sec.axis = sec_axis(trans = transf)
